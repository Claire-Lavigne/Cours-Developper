```bash
# Install django and pip (python) on pc
$ python
$ sudo easy_install pip

$ sudo easy_install django
sudo pip install django --upgrate

# Install django and virtualenv in virtual environment
$ sudo pip install virtualenv
# Move into the wanted paste
$ virtualenv nameofpaste
$ cd nameofpaste
$ source bin/activate
$ pip freeze
$ pip install django
$ pip install django --upgrade
$ django-admin.py stratproject project_name
$ cd project_name
$ python manage.py runserver
```
